protected $guarded = [];

    public static function saveProduct($request, $id=null){

        Product::updateOrCreate(['id' => $id], [

            'name'            => $request->name,
            'category_name'   => $request->category_name,
            'brand_name'      => $request->brand_name,
            'description'     => $request->description,
            'image'           => isset($id) ? (self::getImageUrl($request->file('image'), 'image-upload/product/', Product::find($id)->image )): (self::getImageUrl($request->file('image'), 'image-upload/product/')),
            'status'          => isset($id) ? Product::find($id)->status: 1,
        ]);
    }

    public static function getImageUrl($image, $imageDirectory, $existFileUrl=null)
    {
        if ($image) {
            if (file_exists($existFileUrl))
            {
                unlink($existFileUrl);
            }
            $fileName = time().rand(10,10000).'.'.$image->getClientOriginalExtension();
            $image->move($imageDirectory, $fileName);
            $fileUrl = $imageDirectory.$fileName;
        }
        else {
            if (isset($existFileUrl))
            {
                $fileUrl = $existFileUrl;
            } else {
                $fileUrl = '';
            }
        }
        return $fileUrl;
    }

***********Product Controller*********************
public function createProduct($id=null){
        return view('admin.product.create', ['product' => isset($id) ? Product::find($id) : '']);
    }

    public function newProduct(Request $request, $id=null){

        Product::saveProduct($request, $id);
        if ($id){
            return redirect('/manage-product')->with('message', 'Product Updated Successfully');
        }
        return redirect()->back()->with('message', 'Product Added Successfully');
    }

    public function manageProduct(){
        return view('admin.product.manage', ['products' => Product::all()]);
    }

    public function productStatus($id){

        $product = Product::find($id);
        $product->status = $product->status == 1 ? 0 : 1;
        $product->save();
        return redirect()->back()->with('message', 'Status Changed Successfully');
    }

    public function deleteProduct($id){

        $product = Product::find($id);
        if (file_exists($product->image)){
            unlink($product->image);
        }
        $product->delete();
        return redirect()->back()->with('message', 'Product Delete Successfully');
    }
